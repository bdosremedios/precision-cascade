152 // Perform solve with iterate() scheme updating generic_soln
153 void solve() {
154
155 if (initiated) {
156 throw std::runtime_error(
157 "Can not safely call solve without after a initiation without "
158 "reset"
159 );
160 }
161 initiated = true;
162 // Set res_hist size here since max_iter is mutable before solve
163 initialize_instantiate_residual();
164
165 while (
166 !converged &&
167 ((curr_iter < max_iter) && (get_relres() > target_rel_res))
168 ) {
169 ++curr_iter;
170 iterate();
171 update_residual();
172 if (terminated) { break; }
173 }
174
175 terminated = true;
176 converged = (
177 (get_relres() <= target_rel_res) ||
178 ((curr_iter == 0) &&
179 ((curr_res.norm() /
180 gen_lin_sys_ptr->get_b().norm()).get_scalar() <= target_rel_res))
181 );
182
183 }

200 // Rudimentarily plot relative residual
201 void view_relres_plot(std::string const &arg="normal") const {
202
203 // Get max max_length entries to plot
204 const int max_length(70);
205 std::vector<double> plot_y;
206 if (res_norm_history.size() > max_length) {
207 double h = (res_norm_history.size()-1.0)/(max_length-1.0);
208 for (int i=0; i<max_length; ++i) {
209 plot_y.push_back(res_norm_history[static_cast<int>(i*h)]);
210 }
211 } else {
212 plot_y = res_norm_history;
213 }
214 int length = plot_y.size();
215
216 // Normalize to first residual norm
217 double r_0 = plot_y[0];
218 for (int i=0; i<length; ++i) { plot_y[i] /= r_0; }
219
220 // If want log plot then take log of everything
221 if (arg == "log") {
222 for (int i=0; i<length; ++i) { plot_y[i] = log(plot_y[i])/log(10); }
223 }
224
225 // Find which in height buckets each plot point should be in
226 const int height(12);
227 std::vector<double> plot_y_bucket_index;
228 std::vector<double> bucket_ends;
229 double min_ = *std::min_element(plot_y.cbegin(), plot_y.cend());
230 double max_ = *std::max_element(plot_y.cbegin(), plot_y.cend());
231
232 // Get minimal of target relres and minimum if initiated for bottom of plot
233 if (initiated) {
234 if (arg == "log") {
235 min_ = std::min(min_, log(target_rel_res)/log(10));
236 } else {
237 min_ = std::min(min_, target_rel_res);
238 }
239 }
240
241 double bucket_width = (max_-min_)/static_cast<double>(height);
242 for (double i=1; i<height; ++i) {
243 bucket_ends.push_back(i*bucket_width + min_);
244 }
245 bucket_ends.push_back(max_);
246 for (double y: plot_y) {
247 for (int i=0; i<height; ++i) {
248 if (y <= bucket_ends[i]) {
249 plot_y_bucket_index.push_back(i);
250 break;
251 }
252 }
253 }
254
255 // Iterate over grid and mark typed_soln wherever there is a plot entry
256 const int min_length(3);
257 std::cout << "Display of Relative Residual L-2 Norm: " << std::endl;
258 if (arg == "log") {
259 std::cout << std::setprecision(3) << std::pow(10, max_);
260 } else {
261 std::cout << std::setprecision(3) << max_;
262 };
263 std::cout << " " << std::string(std::max(min_length, length-1), '-')
264 << std::endl;
265 for (int i=height-1; i>=0; --i) {
266 for (int j=-1; j<length; ++j) {
267 if ((j >= 0) && (plot_y_bucket_index[j] == i)) {
268 std::cout << "*";
269 } else {
270 std::cout << " ";
271 }
272 }
273 std::cout << std::endl;
274 }
275 if (arg == "log") {
276 std::cout << std::setprecision(3) << std::pow(10, min_);
277 } else {
278 std::cout << std::setprecision(3) << min_;
279 };
280 std::cout << " " << std::string(std::max(min_length, length-4), '-')
281 << std::endl;
282 std::cout << "Iter: 0"
283 << std::string(std::max(min_length, length-10), ' ')
284 << "Iter: "
