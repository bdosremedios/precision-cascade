152 void update_nextq_and_Hkplus1() {
153
154 int curr_kry_idx = curr_kry_dim-1;
155
156 // Generate next basis vector base
157 next_q = apply_precond_A(
158 Q_kry_basis.get_col(curr_kry_idx).copy_to_vec()

160
161 // Orthogonalize new basis vector with CGS2
162 Vector<TPrecision> first_ortho(
163 Q_kry_basis.transpose_prod_subset_cols(0, curr_kry_dim, next_q)
164 );
165 next_q -= Q_kry_basis.mult_subset_cols(0, curr_kry_dim, first_ortho);
166 Vector<TPrecision> second_ortho(
167 Q_kry_basis.transpose_prod_subset_cols(0, curr_kry_dim, next_q)
168 );
169 next_q -= Q_kry_basis.mult_subset_cols(0, curr_kry_dim, second_ortho);
170
171 H_k.set_slice(0, curr_kry_dim, first_ortho + second_ortho);
172 H_k.set_elem(curr_kry_idx+1, next_q.norm());

176 void update_QR_fact() {
177
178 int curr_kry_idx = curr_kry_dim-1;
179
180 // Initiate next column of QR fact as most recent of H_k with
181 // previous Given's rotations applied
182 H_R.get_col(curr_kry_idx).set_from_vec(H_Q.transpose_prod(H_k));
183
184 // Apply the final Given's rotation manually making H_R upper triangular
185 Scalar<TPrecision> alpha = H_R.get_elem(curr_kry_idx, curr_kry_idx);
186 Scalar<TPrecision> beta = H_R.get_elem(curr_kry_idx+1, curr_kry_idx);
187 Scalar<TPrecision> r = alpha*alpha + beta*beta;
188 r.sqrt();
189 Scalar<TPrecision> c = alpha/r;
190 Scalar<TPrecision> minus_s = beta/r;
191 H_R.set_elem(curr_kry_idx, curr_kry_idx, r);
192 H_R.set_elem(
193 curr_kry_idx+1, curr_kry_idx, SCALAR_ZERO<TPrecision>::get()
194 );
195
196 Vector<TPrecision> H_Q_col_k(
197 H_Q.get_col(curr_kry_idx).copy_to_vec()
198 );
199 Vector<TPrecision> H_Q_col_kp1(
200 H_Q.get_col(curr_kry_idx+1).copy_to_vec()
201 );
202 H_Q.get_col(curr_kry_idx).set_from_vec(
203 H_Q_col_k*c + H_Q_col_kp1*minus_s
204 );
205 H_Q.get_col(curr_kry_idx+1).set_from_vec(
206 H_Q_col_kp1*c - H_Q_col_k*minus_s
207 );
208
209 }
